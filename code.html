                    // get the string from the user
                    function getValues() {
                        let inputString = document.getElementById('inputString').value;
                        let replaceCharacters = '';
                        let inspectedString = inspectCharacters(inputString, replaceCharacters);
                        
                        let stringResult = checkForPalindrome(inspectedString);

                        displayResults(stringResult, inspectedString);
                    }

                    // Inspects and replaces any special characters or spaces in inputString
                    function inspectCharacters(inputString, replaceCharacters) {
                        let lowerCase = inputString.toLowerCase();
                        return lowerCase.replaceAll(/[^a-z0-9]/g, replaceCharacters);
                    }
                    // Check whether a string is a Palindrome or not
                    function checkForPalindrome(inspectedString) {
                        let result = '';
                        let inspect = '';

                        for(let index = inspectedString.length - 1; index >= 0; index-- ) {
                        inspect += inspectedString[index];
                        }

                        if (inspect == inspectedString) {
                            result = "isPalindrome";
                        } else {
                        result = 'notPalindrome';
                        }
                        
                        return result;

                    }

                    // Display the results from checkForPalindrome to the user
                    function displayResults(stringResult, inspectedString) {
                        
                        if(stringResult == 'isPalindrome') {
                        // display a corecct message
                        Swal.fire({
                            icon: 'success',
                            title: 'Nice!',
                            text: `Yes ${inspectedString} is a Palindrome`,
                            background: 'var(--njf-theme-100)',
                            backdrop: false
                        });
                        } else if(stringResult == 'notPalindrome') {
                        // display a false message
                        Swal.fire({
                            icon: 'error',
                            title: 'Sorry!',
                            text: `No, unfortunately ${inspectedString} is not a Palindrome`,
                            background: 'var(--njf-theme-150)',
                            backdrop: false
                        });
                        }
                    }

The code is structured into four functions.

getValues()

The getValues function gets the text from the input element and than sends the input to the function inspectCharacters. When the input is received back from inspectCharacters it is than sent to the function checkForPalindrome. Once the inspection results are received from checkForPalindrome it is sent to the function displayResults so that the information can be displayed.

inspectCharacters()

This function takes the input received from getValues and lowercases all of the capital letters in the string. Than using a regexp and regex replace all special characters and spaces within the String. I than send that information back to getValues.

checkForPalindrome()

The checkForPalindrome function receives the inspected string and than using a for loop and an if else statement check each character in the array and place it into a new scoped variable. I than check the scoped variable to the string that was sent as a parameter to checkForPalindrome, from getValues. If they are the same, they are sent back as a correct Palindrome. If they are not, they are sent back as an incorrect Palindrome.

displayResults()

The displayResults function receives the inspected string and the result of the function checkForPalindrome. Using this I use an if else if statement to check the answer from checkForPalindrome and depending on its answer display a correct alert or an error message letting the user know that it is or is not a Palindrome.
